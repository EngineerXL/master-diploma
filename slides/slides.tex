\section{Актуальность темы}
\begin{frame}
	\frametitle{Актуальность темы}
	\begin{itemize}
		\item Беспилотный транспорт активно развивается: Waymo, <<Автономный транспорт Яндекса>> (роботакси, грузовики, роботы-доставщики).
		\item Лидар обеспечивает геометрию сцены при слабой текстуре и нестабильном GNSS, поэтому лидарная одометрия --- важный компонент локализации.
		\item Крупные динамические объекты (грузовики) перекрывают существенную часть поля зрения лидара, что приводит к деградации оценки собственного движения.
		\item Необходимо сохранять достаточную точность одометрии на коротком участке пути до восстановления наблюдаемости сцены.
	\end{itemize}

\end{frame}


\section{Цель и задачи работы}
\begin{frame}
	\frametitle{Цель и задачи работы}

	\textbf{Объект:} методы локализации и оценки движения беспилотного автомобиля на основе лидарных данных.

	\textbf{Предмет:} алгоритмы лидарной одометрии, учитывающие перекрытие поля зрения при регистрации облаков точек и фильтрации оценок движения.

	\vspace{0.3em}
	\textbf{Цель} --- разработать устойчивый алгоритм лидарной одометрии беспилотного автомобиля, учитывающий перекрытие поля зрения и позволяющий оценивать движение на коротком участке пути.
\end{frame}


\begin{frame}
	\frametitle{Цель и задачи работы}

	\textbf{Задачи:}
	\begin{enumerate}
		\fontsize{10pt}{12pt}\selectfont
		\item Проанализировать существующие подходы к лидарной одометрии и выявить ограничения классических методов регистрации облаков точек.
		\item Описать используемый набор данных Boreas-RT и применяемое подмножество проездов.
		\item Формализовать сценарий перекрытия поля зрения лидара и требования к устойчивости алгоритма на коротком участке пути.
		\item Разработать алгоритм обработки последовательности лидарных кадров: вокселизация, регистрация соседних облаков, оценка преобразования между парами точек с учётом весов, модель движения.
		\item Определить метрики качества и реализовать программный конвейер с возможностью моделирования перекрытия поля зрения.
		\item Провести экспериментальное исследование точности по компонентам линейной и угловой скорости.
	\end{enumerate}
\end{frame}


\section{Постановка задачи}
\begin{frame}
	\frametitle{Постановка задачи}

	Дана последовательность лидарных кадров $\{P_t\}$ и эталонные значения линейных $v_x, v_y, v_z$ и угловых $\omega_x, \omega_y, \omega_z$ скоростей.

	\textbf{Требуется:} построить алгоритм, который по парам соседних облаков $(P_{t-1}, P_t)$ оценивает относительное евклидово преобразование $T_t$ и компоненты скоростей $\hat{\xi}_t$ так, чтобы:

	\begin{itemize}
		\item оценки оставались устойчивыми при частичном перекрытии поля зрения крупным динамическим объектом (грузовиком в соседней полосе);
		\item ошибка оценки не накапливалась катастрофически на коротком интервале (10 секунд $\approx$ 100 кадров);
		\item подавлялось влияние выбросов в соответствиях между точками;
		\item использовалась априорная кинематическая информация о движении автомобиля.
	\end{itemize}
\end{frame}


\section{Сценарий перекрытия поля зрения}
\begin{frame}
	\frametitle{Сценарий перекрытия поля зрения}

	\begin{columns}
		\column{0.55\textwidth}
		Крупный динамический объект (грузовик в соседней полосе) нарушает предпосылки классического ICP:
		\begin{itemize}
			\fontsize{10pt}{12pt}\selectfont
			\item часть статической сцены перестаёт наблюдаться;
			\item уменьшается перекрытие соседних облаков;
			\item появляются точки самого динамического объекта;
			\item оставшаяся часть сцены становится менее информативной.
		\end{itemize}
		Это приводит не только к выбросам, но и к \textbf{дефициту информативных наблюдений}.

\section{Архитектура решения}
\begin{frame}
	\frametitle{Архитектура решения: конвейер обработки кадра}

	Покадровая обработка последовательности лидарных кадров:

	\begin{enumerate}
		\fontsize{11pt}{13pt}\selectfont
		\item \textbf{Предобработка:} удаление точек дальше $r_{\max}$, моделирование грузовика, обгоняющего с относительной скоростью 1~м/с;
		\item \textbf{Двойная вокселизация:} $P_{\mathrm{merge}} = \mathcal{V}_{\alpha v}(P)$, $\hat{P} = \mathcal{V}_{\beta v}(P_{\mathrm{merge}})$;
		\item \textbf{Начальное приближение} по предыдущей оценке движения: $T_{\mathrm{pred}} = \exp(\xi_{t-1}\Delta t)$;
		\item \textbf{Адаптивный порог соответствий} $\tau_t = 3\sigma_t$ по правилу трёх сигм;
		\item \textbf{Взвешенный ICP} с функцией Гемана---Макклюра и алгоритмом Кабша;
		\item \textbf{Фильтр Калмана} с моделью постоянной скорости --- стабилизация начального приближения для следующего шага ICP.
	\end{enumerate}
\end{frame}


\section{Взвешенный ICP}
\begin{frame}
	\frametitle{Взвешенный ICP: функция Гемана---Макклюра}

	На каждой итерации решается задача
	\begin{equation*}
		\Delta T = \arg\min_T \sum_{(s,q)\in C(\tau_t)} \rho\bigl(\|Ts - q\|_2\bigr),
		\qquad
		\rho(e) = \frac{e^2/2}{\sigma_t/3 + e^2}.
	\end{equation*}

	В реализации используются веса:
	\begin{equation*}
		w(e) = \frac{0.5}{\sigma_t/3 + e^2}.
	\end{equation*}

	\begin{itemize}
		\fontsize{10pt}{12pt}\selectfont
		\item Чем больше расстояние между парой точек --- тем меньше её вклад в оценку преобразования.
		\item Применяются веса, дающие устойчивость к выбросам --- обеспечивается \textbf{снижение влияния выбросов}.
		\item Преобразование вычисляется взвешенным алгоритмом Кабша (\texttt{Rotation.align\_vectors} из SciPy).
		\item Архитектура опирается на KISS-ICP, но регистрация выполняется не frame-to-map, а напрямую между двумя соседними облаками.
	\end{itemize}
\end{frame}


\section{Модель движения и фильтр Калмана}
\begin{frame}
	\frametitle{Модель движения и фильтр Калмана}

	\textbf{Модель постоянной скорости.} Между соседними кадрами движение меняется плавно, поэтому начальное приближение для ICP формируется по предыдущей оценке скоростей.

	\textbf{Фильтр Калмана} применяется к шести компонентам скорости $(v_x, v_y, v_z, \omega_x, \omega_y, \omega_z)$ независимо. Параметры выбраны из инженерных соображений:
	\begin{itemize}
		\fontsize{10pt}{12pt}\selectfont
		\item ковариация процесса по линейной скорости: $Q_v = 0.01$ ($\sim$ $1~\text{м/с}^2 \cdot 0.1~\text{с}$);
		\item ковариация измерения по линейной скорости: $R_v = 0.09$ (точность лидара $\sim$ 3 см на интервале 0.1 с);
		\item ковариация процесса по угловой скорости: $Q_\omega = 0.01$;
		\item ковариация измерения по угловой скорости: $R_\omega = 9\cdot 10^{-6}$.
	\end{itemize}

	\textbf{Ключевая идея:} фильтр используется прежде всего для стабилизации начального приближения ICP на следующем шаге, а не как постобработка выходов.
\end{frame}


\section{Экспериментальный конвейер}
\begin{frame}
	\frametitle{Экспериментальный конвейер}

	\textbf{Набор данных:} Boreas-RT --- 10 тестовых проездов (Suburbs, Tunnel East, Skyway, Industrial, Farm и др.).

	\textbf{Две серии экспериментов:}
	\begin{itemize}
		\fontsize{10pt}{12pt}\selectfont
		\item \textbf{Основная серия}: 725 десятисекундных сцен (по 100 кадров) с моделируемым обгоном грузовиком. Цель --- проверить устойчивость алгоритма к перекрытию поля зрения.
		\item \textbf{Контрольная серия}: полные проезды без искусственного перекрытия. Цель --- убедиться, что фильтр Калмана не подавляет естественную динамику автомобиля.
	\end{itemize}

	\textbf{Метрики качества по 6 компонентам скорости:} среднее смещение, стандартное отклонение, MSE, MAE, RMSE, 95\%-квантиль абсолютной ошибки.

	\textbf{Статистическое сравнение конфигураций:} парный T-критерий Стьюдента ($\alpha = 0.05$).
\end{frame}

\section{Оценка результата}
\begin{frame}
	\frametitle{Оценка результата}

	\begin{itemize}
		\fontsize{11pt}{13pt}\selectfont
		\item Разработан устойчивый алгоритм лидарной одометрии, учитывающий перекрытие поля зрения: двойная вокселизация, взвешенный ICP с функцией Гемана---Макклюра, модель постоянной скорости и фильтр Калмана.
		\item Реализован экспериментальный конвейер на наборе Boreas-RT: 725 десятисекундных сцен с моделируемым обгоном грузовика.
		\item На полных проездах фильтр Калмана не подавляет естественную динамику автомобиля.
		\item На коротких сценах с перекрытием поля зрения фильтр Калмана статистически значимо уменьшает MSE, MAE, RMSE и 95\%-квантиль ошибки по компоненте $v_x$ ($-15\dots-70\%$).
		\item Основной эффект --- стабилизация начального приближения для следующего ICP, что снижает риск последовательной деградации регистрации.
		\item \textbf{Ограничения:} геометрическая вырожденность некоторых сцен (туннели, трассы) сохраняется; параметры ковариаций выбраны вручную.
	\end{itemize}
\end{frame}
