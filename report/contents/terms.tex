\newglossaryentry{lidar-device}{
    name={Лидар},
    description={лазерный локатор, который использует лазерные лучи для измерения точных расстояний до объектов и их пространственного расположения}
}

\newglossaryentry{odometry}{
    name={Одометрия},
    description={использование данных сенсоров для оценки перемещения робота}
} 

\newglossaryentry{lidar-odometry}{
    name={Лидарная одометрия},
    description={метод оценки положения и ориентации робота на основе данных с лидаров}
}

\newglossaryentry{localization}{
    name={Локализация},
    description={процесс определения положения робота (позы и ориентации) в пространстве (фиксированной системе координат)}
}

\newglossaryentry{correspondence}{
    name={Соответствие},
    description={сопоставление точек между исходным и целевым облаками точек по критерию ближайшего соседа}
}

\newglossaryentry{dynamic-system}{
    name={Динамическая система},
    description={математическую модель некоторого объекта, процесса или явления, для которого однозначно определено понятие состояния как совокупности некоторых величин в данный момент времени и задан закон, который описывает изменение начального состояния с течением времени}
}

\newglossaryentry{lidar-point-cloud}{
    name={Лидарное облако точек},
    description={совокупность трёхмерных точек, полученных по измерениям лидара и описывающих пространственную структуру наблюдаемой сцены}
}

\newglossaryentry{voxelization}{
    name={Вокселизация},
    description={способ дискретизации трёхмерного пространства, при котором оно разбивается на объёмные ячейки фиксированного размера, называемые вокселями}
}

\newglossaryentry{euclidean-transform}{
    name={Евклидово преобразование},
    description={преобразование пространства, сохраняющее расстояния между точками и включающее поворот, перенос или их композицию}
}

\newglossaryentry{z-score}{
    name={Z-оценка},
    description={статистическая величина, показывающая, на сколько стандартных отклонений значение наблюдаемой величины отличается от среднего значения выборки}
}

\newglossaryentry{p-value}{
    name={P-значение},
    description={вероятность получить наблюдаемый или более экстремальный результат при условии, что нулевая гипотеза верна}
}
