\newglossaryentry{lidar-device}{
    name={Лидар},
    description={лазерный локатор, который использует лазерные лучи для измерения точных расстояний до объектов и их пространственного расположения}
}

\newglossaryentry{lidar-odometry}{
    name={Лидарная одометрия},
    description={метод оценки положения и ориентации робота на основе данных с лидаров}
}

\newglossaryentry{dynamic-system}{
    name={Динамическая система},
    description={математическую модель некоторого объекта, процесса или явления, для которого однозначно определено понятие состояния как совокупности некоторых величин в данный момент времени и задан закон, который описывает изменение начального состояния с течением времени}
}
